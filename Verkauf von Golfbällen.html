Alles klar, du willst eine eigene Website, um gebrauchte Golfbälle zu verkaufen? Kein Problem! Ich kann dir einen einfachen Beispielcode für so eine Website schreiben — eine Startseite mit Produktlisten und Verkaufsmöglichkeit.

Hier ein minimalistisches Beispiel in HTML und etwas JavaScript:

html
<!DOCTYPE html>
<html lang="de">
<head>
  <meta charset="UTF-8" />
  <meta name="viewport" content="width=device-width, initial-scale=1" />
  <title>Gebrauchte Golfbälle verkaufen</title>
  <style>
    body {
      font-family: Arial, sans-serif;
      max-width: 800px;
      margin: 2rem auto;
      padding: 1rem;
      background: #f9f9f9;
    }
    h1 {
      color: #006400;
    }
    .ball-list {
      margin-top: 2rem;
    }
    .ball {
      background: white;
      padding: 1rem;
      margin-bottom: 1rem;
      border-radius: 6px;
      box-shadow: 0 0 5px #ccc;
      display: flex;
      justify-content: space-between;
      align-items: center;
    }
    button {
      background-color: #006400;
      color: white;
      border: none;
      padding: 0.5rem 1rem;
      border-radius: 4px;
      cursor: pointer;
    }
    button:hover {
      background-color: #004d00;
    }
    #form-section {
      margin-top: 3rem;
      background: white;
      padding: 1rem;
      border-radius: 6px;
      box-shadow: 0 0 5px #ccc;
    }
    label {
      display: block;
      margin-top: 1rem;
    }
    input[type="text"],
    input[type="number"] {
      width: 100%;
      padding: 0.5rem;
      margin-top: 0.3rem;
      border: 1px solid #ccc;
      border-radius: 4px;
    }
  </style>
</head>
<body>
  <h1>Gebrauchte Golfbälle verkaufen</h1>

  <div class="ball-list" id="ball-list">
    <!-- Liste der Golfbälle -->
  </div>

  <section id="form-section">
    <h2>Neuen Golfball hinzufügen</h2>
    <form id="sell-form">
      <label for="brand">Marke:</label>
      <input type="text" id="brand" name="brand" required />

      <label for="condition">Zustand:</label>
      <input type="text" id="condition" name="condition" placeholder="z.B. gebraucht, neu" required />

      <label for="price">Preis (EUR):</label>
      <input type="number" id="price" name="price" min="0" step="0.01" required />

      <button type="submit">Golfball zum Verkauf hinzufügen</button>
    </form>
  </section>

  <script>
    const ballListEl = document.getElementById('ball-list');
    const form = document.getElementById('sell-form');

    // Anfangs leere Liste
    let balls = [];

    // Funktion um die Liste anzuzeigen
    function renderBalls() {
      if (balls.length === 0) {
        ballListEl.innerHTML = '<p>Keine Golfbälle zum Verkauf.</p>';
        return;
      }
      ballListEl.innerHTML = '';
      balls.forEach((ball, index) => {
        const ballEl = document.createElement('div');
        ballEl.className = 'ball';
        ballEl.innerHTML = `
          <div>
            <strong>${ball.brand}</strong> - Zustand: ${ball.condition} - Preis: €${ball.price.toFixed(2)}
          </div>
          <button onclick="buyBall(${index})">Kaufen</button>
        `;
        ballListEl.appendChild(ballEl);
      });
    }

    // "Kaufen"-Button Funktion
    function buyBall(index) {
      alert(`Danke für deinen Kauf von ${balls[index].brand} Golfball zum Preis von €${balls[index].price.toFixed(2)}!`);
      balls.splice(index, 1);
      renderBalls();
    }

    // Formular absenden
    form.addEventListener('submit', (event) => {
      event.preventDefault();
      const brand = form.brand.value.trim();
      const condition = form.condition.value.trim();
      const price = parseFloat(form.price.value);

      if (brand && condition && price >= 0) {
        balls.push({ brand, condition, price });
        form.reset();
        renderBalls();
      }
    });

    renderBalls();
  </script>
</body>
</html>

Du kannst diese Datei einfach als index.html speichern und im Browser öffnen. Damit hast du eine kleine Verkaufsseite, wo du deine gebrauchten Golfbälle eintragen und „verkaufen“ kannst.
